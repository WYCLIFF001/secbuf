2//! Statistics tracking for buffer pools.
3
4/// Statistics for standard buffer pool.
5#[derive(Debug, Clone)]
6pub struct PoolStats {
7    /// Number of buffers currently available in the pool
8    pub available: usize,
9    /// Total number of buffers allocated since pool creation
10    pub total_allocated: usize,
11    /// Total number of acquire() calls
12    pub total_acquired: usize,
13    /// Total number of buffers returned to pool
14    pub total_returned: usize,
15    /// Size of each buffer in bytes
16    pub buffer_size: usize,
17    /// Maximum number of buffers the pool can hold
18    pub max_pool_size: usize,
19}
20
21impl PoolStats {
22    /// Returns the number of buffers currently in use (acquired but not returned).
23    ///
24    /// # Examples
25    ///
26    /// ```
27    /// use buffer::prelude::*;
28    ///
29    /// let pool = BufferPool::new(PoolConfig::default());
30    /// let _buf = pool.acquire();
31    ///
32    /// let stats = pool.stats();
33    /// assert_eq!(stats.in_use(), 1);
34    /// ```
35    pub fn in_use(&self) -> usize {
36        self.total_acquired.saturating_sub(self.total_returned)
37    }
38
39    /// Returns the pool hit rate as a percentage (0.0-100.0).
40    ///
41    /// A higher hit rate indicates better buffer reuse and fewer allocations.
42    ///
43    /// # Examples
44    ///
45    /// ```
46    /// use buffer::prelude::*;
47    ///
48    /// let pool = BufferPool::new(PoolConfig::default());
49    /// for _ in 0..10 {
50    ///     let _buf = pool.acquire();
51    /// }
52    ///
53    /// let stats = pool.stats();
54    /// assert!(stats.hit_rate() >= 0.0);
55    /// assert!(stats.hit_rate() <= 100.0);
56    /// ```
57    pub fn hit_rate(&self) -> f64 {
58        if self.total_acquired == 0 {
59            return 0.0;
60        }
61        let reused = self.total_acquired.saturating_sub(self.total_allocated);
62        (reused as f64 / self.total_acquired as f64) * 100.0
63    }
64}

66/// Statistics for fast buffer pool with thread-local caching.
67#[derive(Debug, Clone)]
68pub struct FastPoolStats {
69    /// Number of buffers currently available in the global pool
70    pub available: usize,
71    /// Total number of new buffers allocated
72    pub allocated: usize,
73    /// Total number of acquire() calls
74    pub acquired: usize,
75    /// Total number of buffers returned
76    pub returned: usize,
77    /// Number of times a buffer was acquired from thread-local cache
78    pub cache_hits: usize,
79    /// Number of buffers lost when threads exit with cached buffers
80    pub thread_local_lost: usize,
81}
82
83impl FastPoolStats {
84    /// Returns the number of buffers currently in use.
85    ///
86    /// # Examples
87    ///
88    /// ```
89    /// use buffer::prelude::*;
90    ///
91    /// let pool = FastBufferPool::new(PoolConfig::default());
92    /// let _buf = pool.acquire();
93    ///
94    /// let stats = pool.stats();
95    /// assert_eq!(stats.in_use(), 1);
96    /// ```
97    pub fn in_use(&self) -> usize {
98        self.acquired.saturating_sub(self.returned)
99    }
100
101    /// Returns the thread-local cache hit rate as a percentage.
102    ///
103    /// A higher cache hit rate indicates better thread-local cache utilization.
104    ///
105    /// # Examples
106    ///
107    /// ```
108    /// use buffer::prelude::*;
109    ///
110    /// let pool = FastBufferPool::new(PoolConfig::default());
111    /// for _ in 0..10 {
112    ///     let _buf = pool.acquire();
113    /// }
114    ///
115    /// let stats = pool.stats();
116    /// assert!(stats.cache_hit_rate() >= 0.0);
117    /// assert!(stats.cache_hit_rate() <= 100.0);
118    /// ```
119    pub fn cache_hit_rate(&self) -> f64 {
120        if self.acquired == 0 {
121            return 0.0;
122        }
123        (self.cache_hits as f64 / self.acquired as f64) * 100.0
124    }
125
126    /// Returns the overall pool hit rate (cache + global pool) as a percentage.
127    ///
128    /// This indicates how often buffers were reused vs freshly allocated.
129    ///
130    /// # Examples
131    ///
132    /// ```
133    /// use buffer::prelude::*;
134    ///
135    /// let pool = FastBufferPool::new(PoolConfig::default());
136    /// for _ in 0..10 {
137    ///     let _buf = pool.acquire();
138    /// }
139    ///
140    /// let stats = pool.stats();
141    /// assert!(stats.pool_hit_rate() >= 0.0);
142    /// assert!(stats.pool_hit_rate() <= 100.0);
143    /// ```
144    pub fn pool_hit_rate(&self) -> f64 {
145        if self.acquired == 0 {
146            return 0.0;
147        }
148        let hits = self.acquired.saturating_sub(self.allocated);
149        (hits as f64 / self.acquired as f64) * 100.0
150    }
151
152    /// Estimates total leaked memory from dead threads.
153    ///
154    /// Returns the number of bytes lost to thread-local caches when threads exit.
155    ///
156    /// # Arguments
157    ///
158    /// * `buffer_size` - The size of each buffer in bytes
159    ///
160    /// # Examples
161    ///
162    /// ```
163    /// use buffer::prelude::*;
164    ///
165    /// let pool = FastBufferPool::new(PoolConfig {
166    ///     buffer_size: 8192,
167    ///     max_pool_size: 100,
168    ///     min_pool_size: 10,
169    /// });
170    ///
171    /// let stats = pool.stats();
172    /// let leaked_bytes = stats.leaked_bytes(8192);
173    /// assert_eq!(leaked_bytes, stats.thread_local_lost * 8192);
174    /// ```
175    pub fn leaked_bytes(&self, buffer_size: usize) -> usize {
176        self.thread_local_lost * buffer_size
177    }
178
179    /// Checks if the leak rate is concerning.
180    ///
181    /// Returns `true` if more than 10% of allocated buffers have been lost
182    /// to dead threads.
183    ///
184    /// # Examples
185    ///
186    /// ```
187    /// use buffer::prelude::*;
188    ///
189    /// let pool = FastBufferPool::new(PoolConfig::default());
190    /// let stats = pool.stats();
191    ///
192    /// if stats.has_leak_concern() {
193    ///     eprintln!("Warning: High buffer leak rate detected!");
194    /// }
195    /// ```
196    pub fn has_leak_concern(&self) -> bool {
197        if self.allocated == 0 {
198            return false;
199        }
200        let leak_rate = (self.thread_local_lost as f64 / self.allocated as f64) * 100.0;
201        leak_rate > 10.0
202    }
203}
