2//! Standard mutex-based buffer pool
3//!
4//! # Pool Size Limits
5//!
6//! When buffers are returned and the pool is at `max_pool_size`:
7//! - The buffer is securely zeroed (via zeroize crate)
8//! - The buffer is dropped and not retained in the pool
9//! - This prevents unbounded memory growth
10//!
11//! # Memory Safety
12//!
13//! All buffers are automatically and securely zeroed on drop using the zeroize
14//! crate, which provides compiler-resistant memory clearing.
15
16use super::config::PoolConfig;
17use super::stats::PoolStats;
18use crate::buffer::Buffer;
19use std::sync::{Arc, Mutex};
20
21pub(crate) struct PoolInner {
22    pub(crate) buffers: Vec<Buffer>,
23    pub(crate) config: PoolConfig,
24    pub(crate) total_allocated: usize,
25    pub(crate) total_acquired: usize,
26    pub(crate) total_returned: usize,
27}

29/// Standard thread-safe buffer pool using Mutex.
30///
31/// Simple and reliable implementation suitable for most applications.
32/// For maximum performance in high-contention scenarios, use [`crate::pool::FastBufferPool`].
33///
34/// # Thread Safety
35///
36/// This pool uses a `Mutex` for synchronization and can be safely shared across
37/// threads using `Arc`.
38///
39/// # Performance Characteristics
40///
41/// - Acquire: O(1) amortized
42/// - Return: O(1) amortized
43/// - Mutex contention in high-concurrency scenarios
44///
45/// For lock-free operation, consider [`crate::pool::FastBufferPool`].
46///
47/// # Example
48///
49/// ```rust
50/// use secbuf::prelude::*;
51///
52/// let pool = BufferPool::new(PoolConfig {
53///     buffer_size: 4096,
54///     max_pool_size: 100,
55///     min_pool_size: 10,
56/// });
57///
58/// let mut buf = pool.acquire();
59/// buf.put_u32(42)?;
60/// // Buffer automatically returned on drop
61/// # Ok::<(), secbuf::BufferError>(())
62/// ```
63pub struct BufferPool {
64    pub(crate) inner: Arc<Mutex<PoolInner>>,
65}
66
67impl Default for BufferPool {
68    fn default() -> Self {
69        Self::new(PoolConfig::default())
70    }
71}
72
73impl BufferPool {
74    /// Creates a new buffer pool with the given configuration.
75    ///
76    /// Pre-warms the pool with `config.min_pool_size` buffers.
77    pub fn new(config: PoolConfig) -> Self {
78        let mut buffers = Vec::with_capacity(config.min_pool_size);
79
80        // Pre-warm the pool
81        for _ in 0..config.min_pool_size {
82            buffers.push(Buffer::new(config.buffer_size));
83        }
84
85        Self {
86            inner: Arc::new(Mutex::new(PoolInner {
87                buffers,
88                config,
89                total_allocated: 0,
90                total_acquired: 0,
91                total_returned: 0,
92            })),
93        }
94    }
95
96    /// Acquires a buffer from the pool.
97    ///
98    /// If no buffers are available, allocates a new one.
99    /// The buffer is automatically returned to the pool when dropped.
100    ///
101    /// # Returns
102    ///
103    /// A `PooledBuffer` that will be automatically returned to the pool when dropped.
104    pub fn acquire(&self) -> PooledBuffer {
105        let mut inner = self.inner.lock().unwrap();
106        inner.total_acquired += 1;
107
108        let buffer = inner.buffers.pop().unwrap_or_else(|| {
109            inner.total_allocated += 1;
110            Buffer::new(inner.config.buffer_size)
111        });
112
113        PooledBuffer {
114            buffer: Some(buffer),
115            pool: Arc::clone(&self.inner),
116        }
117    }
118
119    /// Returns the number of available buffers in the pool.
120    pub fn available(&self) -> usize {
121        self.inner.lock().unwrap().buffers.len()
122    }
123
124    /// Returns pool statistics.
125    pub fn stats(&self) -> PoolStats {
126        let inner = self.inner.lock().unwrap();
127        PoolStats {
128            available: inner.buffers.len(),
129            total_allocated: inner.total_allocated,
130            total_acquired: inner.total_acquired,
131            total_returned: inner.total_returned,
132            buffer_size: inner.config.buffer_size,
133            max_pool_size: inner.config.max_pool_size,
134        }
135    }
136
137    /// Shrinks the pool to minimum size.
138    ///
139    /// Removes excess buffers beyond `min_pool_size`. Removed buffers are
140    /// securely zeroed before being dropped.
141    pub fn shrink(&self) {
142        let mut inner = self.inner.lock().unwrap();
143        let min_size = inner.config.min_pool_size;
144        inner.buffers.truncate(min_size);
145        inner.buffers.shrink_to_fit();
146    }
147
148    /// Clears all buffers from the pool.
149    ///
150    /// All cleared buffers are securely zeroed before being dropped.
151    pub fn clear(&self) {
152        let mut inner = self.inner.lock().unwrap();
153        inner.buffers.clear();
154    }
155
156    /// Grows the pool to the specified size.
157    ///
158    /// Allocates new buffers and adds them to the pool up to the
159    /// lesser of `target_size` or `config.max_pool_size`.
160    pub fn grow(&self, target_size: usize) {
161        let mut inner = self.inner.lock().unwrap();
162        let max_size = inner.config.max_pool_size;
163        let buffer_size = inner.config.buffer_size;
164        let target = target_size.min(max_size);
165
166        while inner.buffers.len() < target {
167            inner.buffers.push(Buffer::new(buffer_size));
168        }
169    }
170}

172/// A buffer borrowed from the standard pool.
173///
174/// Automatically returned to the pool when dropped, unless `leak()` or `drop_now()` is called.
175///
176/// # Automatic Cleanup
177///
178/// When dropped normally, the buffer is:
179/// 1. Reset (position and length cleared)
180/// 2. Returned to the pool if space available
181/// 3. Otherwise, securely zeroed and dropped (not retained in pool)
182///
183/// To prevent the buffer from being returned to the pool, use `leak()` or `drop_now()`.
184pub struct PooledBuffer {
185    pub(crate) buffer: Option<Buffer>,
186    pub(crate) pool: Arc<Mutex<PoolInner>>,
187}
188
189impl PooledBuffer {
190    /// Leaks the buffer, preventing return to pool.
191    ///
192    /// Returns the owned buffer. The caller is responsible for eventual cleanup.
193    /// The buffer will still be securely zeroed when finally dropped due to the
194    /// `#[zeroize(drop)]` attribute on `Buffer`.
195    ///
196    /// # Example
197    ///
198    /// ```rust
199    /// use secbuf::prelude::*;
200    ///
201    /// let pool = BufferPool::new(PoolConfig::default());
202    /// let pooled_buf = pool.acquire();
203    /// let owned_buf = pooled_buf.leak();
204    /// // owned_buf is now independent of the pool
205    /// ```
206    pub fn leak(mut self) -> Buffer {
207        self.buffer.take().unwrap()
208    }
209
210    /// Immediately drops the buffer without returning it to the pool.
211    ///
212    /// The buffer is securely zeroed before being dropped.
213    /// This is useful when you want to ensure a buffer is not reused.
214    ///
215    /// # Example
216    ///
217    /// ```rust
218    /// use secbuf::prelude::*;
219    ///
220    /// let pool = BufferPool::new(PoolConfig::default());
221    /// let mut buf = pool.acquire();
222    /// buf.put_bytes(b"sensitive data").unwrap();
223    /// buf.drop_now(); // Securely erased, not returned to pool
224    /// ```
225    pub fn drop_now(mut self) {
226        if let Some(mut buffer) = self.buffer.take() {
227            buffer.burn(); // Explicit secure erasure
228            drop(buffer);
229        }
230    }
231
232    /// Returns the buffer's capacity.
233    #[inline]
234    pub fn capacity(&self) -> usize {
235        self.buffer.as_ref().unwrap().capacity()
236    }
237}
238
239impl std::ops::Deref for PooledBuffer {
240    type Target = Buffer;
241
242    fn deref(&self) -> &Self::Target {
243        self.buffer.as_ref().unwrap()
244    }
245}
246
247impl std::ops::DerefMut for PooledBuffer {
248    fn deref_mut(&mut self) -> &mut Self::Target {
249        self.buffer.as_mut().unwrap()
250    }
251}
252
253impl Drop for PooledBuffer {
254    /// Automatically returns the buffer to the pool.
255    ///
256    /// The buffer is reset and returned to the pool if space is available.
257    /// If the pool is at max capacity, the buffer is securely zeroed and dropped.
258    fn drop(&mut self) {
259        if let Some(mut buffer) = self.buffer.take() {
260            buffer.reset();
261
262            let mut inner = self.pool.lock().unwrap();
263            inner.total_returned += 1;
264
265            if inner.buffers.len() < inner.config.max_pool_size {
266                inner.buffers.push(buffer);
267            }
268            // else: buffer is dropped here and securely zeroed via #[zeroize(drop)]
269        }
270    }
271}

273#[cfg(test)]
274mod tests {
275    use super::*;
276
277    #[test]
278    fn test_pool_basic() {
279        let pool = BufferPool::new(PoolConfig {
280            buffer_size: 1024,
281            max_pool_size: 10,
282            min_pool_size: 2,
283        });
284
285        assert_eq!(pool.available(), 2);
286
287        let _buf = pool.acquire();
288        let stats = pool.stats();
289        assert_eq!(stats.total_acquired, 1);
290    }
291
292    #[test]
293    fn test_drop_now() {
294        let pool = BufferPool::new(PoolConfig {
295            buffer_size: 1024,
296            max_pool_size: 10,
297            min_pool_size: 2,
298        });
299
300        let initial_available = pool.available();
301
302        {
303            let mut buf = pool.acquire();
304            buf.put_u32(12345).unwrap();
305            buf.drop_now(); // Should not return to pool
306        }
307
308        // Pool should not have gained a buffer back
309        assert_eq!(pool.available(), initial_available - 1);
310    }
311
312    #[test]
313    fn test_leak() {
314        let pool = BufferPool::new(PoolConfig {
315            buffer_size: 1024,
316            max_pool_size: 10,
317            min_pool_size: 2,
318        });
319
320        let pooled_buf = pool.acquire();
321        let _owned_buf = pooled_buf.leak();
322
323        let stats = pool.stats();
324        assert_eq!(stats.total_acquired, 1);
325        // Leaked buffer doesn't return to pool
326    }
327
328    #[test]
329    fn test_normal_return() {
330        let pool = BufferPool::new(PoolConfig {
331            buffer_size: 1024,
332            max_pool_size: 10,
333            min_pool_size: 2,
334        });
335
336        let initial_available = pool.available();
337
338        {
339            let _buf = pool.acquire();
340        } // Normal drop - should return to pool
341
342        assert_eq!(pool.available(), initial_available);
343    }
