2//! Core buffer structure with aggressive memory management
3//!
4//! This module provides the fundamental [`Buffer`] type with position tracking,
5//! automatic secure memory zeroing on drop, and aggressive memory cleanup capabilities
6//! to prevent memory spikes and unbounded growth in long-running services.
7//!
8//! # Memory Management Strategy
9//!
10//! The buffer provides three levels of cleanup:
11//! 1. **Automatic**: Memory is securely zeroed on drop (via `#[zeroize(drop)]`)
12//! 2. **Periodic**: Reset and trim oversized buffers during idle periods
13//! 3. **Aggressive**: Force shrink when memory pressure is detected
14//!
15//! # Examples
16//!
17//! ```
18//! use secbuf::Buffer;
19//! # use secbuf::BufferError;
20//!
21//! // Basic usage
22//! let mut buf = Buffer::new(1024);
23//! buf.put_u32(42)?;
24//! buf.put_bytes(b"hello")?;
25//!
26//! // Detect waste
27//! if buf.is_wasteful() {
28//!     println!("Buffer has {} bytes overhead", buf.memory_overhead());
29//! }
30//!
31//! // Aggressive cleanup when needed
32//! buf.aggressive_shrink(512); // Keep only 512 bytes capacity
33//! # Ok::<(), BufferError>(())
34//! ```

45/// A high-performance linear buffer with aggressive memory management.
46///
47/// The buffer automatically and securely zeros its memory on drop using
48/// the [`zeroize`] crate, which provides compiler-resistant memory clearing.
49///
50/// # Memory Safety
51///
52/// All buffer memory is automatically zeroed when the buffer is dropped,
53/// preventing sensitive data from remaining in memory. Additional cleanup
54/// methods provide explicit control for long-running services.
55///
56/// # Memory Management
57///
58/// The buffer provides several cleanup strategies:
59///
60/// - [`reset`](Self::reset): Clear position/length without zeroing (fast, for reuse)
61/// - [`burn`](Self::burn): Securely zero all memory
62/// - [`burn_and_free_memory`](Self::burn_and_free_memory): Zero and release capacity
63/// - [`aggressive_shrink`](Self::aggressive_shrink): Force capacity reduction
64/// - [`reset_and_trim`](Self::reset_and_trim): Conditional shrinking for efficiency
65///
66/// # Waste Detection
67///
68/// Use [`is_wasteful`](Self::is_wasteful) and [`memory_overhead`](Self::memory_overhead)
69/// to detect buffers with excessive unused capacity that should be shrunk.
70///
71/// # Examples
72///
73/// ```
74/// use secbuf::Buffer;
75/// # use secbuf::BufferError;
76///
77/// let mut buf = Buffer::try_new(1024)?;
78/// buf.put_u32(42)?;
79/// buf.put_bytes(b"hello")?;
80///
81/// // Check for waste
82/// if buf.is_wasteful() {
83///     // Trim to reasonable size
84///     buf.reset_and_trim(512);
85/// }
86/// # Ok::<(), BufferError>(())
87/// ```
88#[derive(Zeroize)]
89#[zeroize(drop)]

92    pub(crate) data: Vec<u8>,

94    pub pos: usize,

97}

100    /// Creates a new buffer with zeroed memory.
101    ///
102    /// This allocates a `Vec<u8>` of the specified size and initializes it with zeros.
103    /// The allocation is eager (not lazy) - all memory is allocated immediately.
104    ///
105    /// # Errors
106    ///
107    /// Returns [`BufferError::SizeTooBig`] if `size` exceeds [`BUF_MAX_SIZE`] (1GB).
108    ///
109    /// # Examples
110    ///
111    /// ```
112    /// use secbuf::Buffer;
113    ///
114    /// let buf = Buffer::try_new(8192)?;
115    /// assert_eq!(buf.capacity(), 8192);
116    /// assert_eq!(buf.len(), 0);
117    /// # Ok::<(), secbuf::BufferError>(())
118    /// ```
119    #[inline]
120    pub fn try_new(size: usize) -> Result<Self> {
121        if size > BUF_MAX_SIZE {
122            return Err(BufferError::SizeTooBig);
123        }
124        Ok(Self {
125            data: vec![0; size],
126            pos: 0,
127            len: 0,
128        })
129    }

151    /// Creates a new buffer with pre-allocated capacity but zero length.
152    ///
153    /// The internal `Vec` is allocated with the specified capacity but has zero length.
154    /// You must call [`set_len`](Self::set_len) or use write operations to extend the
155    /// valid data region.
156    ///
157    /// # Errors
158    ///
159    /// Returns [`BufferError::SizeTooBig`] if `capacity` exceeds [`BUF_MAX_SIZE`] (1GB).
160    ///
161    /// # Examples
162    ///
163    /// ```
164    /// use secbuf::Buffer;
165    /// # use secbuf::BufferError;
166    ///
167    /// let mut buf = Buffer::try_with_capacity(8192)?;
168    /// assert_eq!(buf.capacity(), 8192);
169    /// assert_eq!(buf.len(), 0);
170    ///
171    /// // Must set length before writing to ensure capacity exists
172    /// buf.set_len(100)?;
173    /// buf.set_pos(0)?;
174    /// buf.put_u32(42)?;
175    /// assert_eq!(buf.len(), 100);
176    /// # Ok::<(), BufferError>(())
177    /// ```
178    #[inline]
179    pub fn try_with_capacity(capacity: usize) -> Result<Self> {
180        if capacity > BUF_MAX_SIZE {
181            return Err(BufferError::SizeTooBig);
182        }
183        let mut data = vec![0; capacity];
184        // Initialize to capacity with zeros so the memory exists
185        data.resize(capacity, 0);
186        Ok(Self {
187            data,
188            pos: 0,
189            len: 0,
190        })
191    }

397    /// Increments the position by `n` bytes.
398    ///
399    /// # Errors
400    ///
401    /// Returns [`BufferError::IncrementTooLarge`] if `n` exceeds max increment.
402    /// Returns [`BufferError::PositionOutOfBounds`] if new position exceeds length.
403    ///
404    /// # Examples
405    ///
406    /// ```
407    /// use secbuf::Buffer;
408    /// # use secbuf::BufferError;
409    ///
410    /// let mut buf = Buffer::new(1024);
411    /// buf.put_u32(42)?;
412    /// buf.set_pos(0)?;
413    /// buf.incr_pos(4)?;
414    /// assert_eq!(buf.pos(), 4);
415    /// # Ok::<(), BufferError>(())
416    /// ```
417    #[inline]
418    pub fn incr_pos(&mut self, n: usize) -> Result<()> {
419        if n > BUF_MAX_INCR {
420            return Err(BufferError::IncrementTooLarge);
421        }
422        let new_pos = self.pos.saturating_add(n);
423        if new_pos > self.len {
424            return Err(BufferError::PositionOutOfBounds);
425        }
426        self.pos = new_pos;
427        Ok(())
428    }

430    /// Increments the length by `n` bytes.
431    ///
432    /// # Errors
433    ///
434    /// Returns [`BufferError::IncrementTooLarge`] if `n` exceeds max increment.
435    /// Returns [`BufferError::PositionOutOfBounds`] if new length exceeds capacity.
436    ///
437    /// # Examples
438    ///
439    /// ```
440    /// use secbuf::Buffer;
441    /// # use secbuf::BufferError;
442    ///
443    /// let mut buf = Buffer::new(1024);

445    /// assert_eq!(buf.len(), 100);
446    /// # Ok::<(), BufferError>(())
447    /// ```
448    #[inline]
449    pub fn incr_len(&mut self, n: usize) -> Result<()> {
450        if n > BUF_MAX_INCR {
451            return Err(BufferError::IncrementTooLarge);
452        }
453        let new_len = self.len.saturating_add(n);
454        if new_len > self.data.len() {
455            return Err(BufferError::PositionOutOfBounds);
456        }
457        self.len = new_len;
458        Ok(())
459    }

579    /// Aggressive cleanup: zeros memory AND shrinks capacity to zero.
580    ///
581    /// Use when buffer won't be reused and you want to free memory immediately.
582    /// This is more aggressive than [`burn`](Self::burn) which keeps capacity.
583    ///
584    /// # Use Cases
585    ///
586    /// - Connection termination in long-running services
587    /// - Memory pressure situations
588    /// - One-time buffer usage
589    ///
590    /// # Examples
591    ///
592    /// ```
593    /// use secbuf::Buffer;
594    /// # use secbuf::BufferError;
595    ///
596    /// let mut buf = Buffer::new(8192);
597    /// buf.put_bytes(b"data")?;
598    ///
599    /// buf.burn_and_free_memory();
600    ///
601    /// assert_eq!(buf.len(), 0);
602    /// assert_eq!(buf.capacity(), 0); // Memory freed
603    /// # Ok::<(), BufferError>(())

605    pub fn burn_and_free_memory(&mut self) {
606        self.data.zeroize();
607        self.data.clear();
608        self.data.shrink_to_fit();
609        self.pos = 0;
610        self.len = 0;
611    }

613    /// Resets and aggressively shrinks if capacity is much larger than needed.
614    ///
615    /// Useful for long-lived buffers that had temporary spikes. This method
616    /// intelligently shrinks only when wasteful (capacity >4x target AND >8KB).
617    ///
618    /// # Parameters
619    ///
620    /// - `target_size`: Desired capacity after shrinking
621    ///
622    /// # Shrinking Criteria
623    ///
624    /// Shrinks if:
625    /// - capacity > target_size * 4 AND
626    /// - capacity > 8192 bytes
627    ///
628    /// # Use Cases
629    ///
630    /// - Periodic cleanup in idle connections
631    /// - After processing large messages
632    /// - Memory pressure mitigation
633    ///
634    /// # Examples
635    ///
636    /// ```
637    /// use secbuf::Buffer;
638    /// # use secbuf::BufferError;
639    ///
640    /// let mut buf = Buffer::new(65536); // 64KB
641    /// buf.put_bytes(b"small")?;
642    ///
643    /// // Shrinks because 64KB >> 1KB * 4
644    /// buf.reset_and_trim(1024);
645    ///
646    /// assert!(buf.capacity() <= 4096);
647    /// assert_eq!(buf.len(), 0);
648    /// # Ok::<(), BufferError>(())
649    /// ```
650    pub fn reset_and_trim(&mut self, target_size: usize) {
651        self.reset();
652
653        let cap = self.data.capacity();
654        let should_shrink = cap > target_size * 4 && cap > 8192;
655
656        if should_shrink {
657            // Zero current data first
658            self.data.zeroize();
659
660            // Resize to target
661            self.data.resize(target_size, 0);
662            self.data.shrink_to_fit();
663        }
664    }

666    /// Aggressive shrink: zeros everything and shrinks to minimal size.
667    ///
668    /// Forces capacity reduction regardless of current usage. Use when
669    /// memory pressure is detected or buffer won't be used for a while.
670    ///
671    /// # Parameters
672    ///
673    /// - `keep_size`: Bytes of capacity to retain (0 = free all)
674    ///
675    /// # Examples
676    ///
677    /// ```
678    /// use secbuf::Buffer;
679    /// # use secbuf::BufferError;
680    ///
681    /// let mut buf = Buffer::new(8192);
682    /// buf.put_bytes(b"data")?;
683    ///
684    /// // Free all memory
685    /// buf.aggressive_shrink(0);
686    /// assert_eq!(buf.capacity(), 0);
687    ///
688    /// // Or keep minimal capacity
689    /// let mut buf2 = Buffer::new(8192);
690    /// buf2.aggressive_shrink(1024);
691    /// assert_eq!(buf2.capacity(), 1024);
692    /// # Ok::<(), BufferError>(())
693    /// ```
694    pub fn aggressive_shrink(&mut self, keep_size: usize) {
695        // Zero all data
696        self.data.zeroize();
697
698        // Reset state
699        self.pos = 0;
700        self.len = 0;
701
702        // Shrink to requested size
703        if keep_size == 0 {
704            self.data.clear();
705            self.data.shrink_to_fit();
706        } else {
707            self.data.resize(keep_size, 0);
708            self.data.shrink_to_fit();
709        }
710    }

739    /// Returns memory overhead in bytes (capacity - length).
740    ///
741    /// This is the amount of allocated but unused memory. High overhead
742    /// indicates opportunities for shrinking.
743    ///
744    /// # Examples
745    ///
746    /// ```
747    /// use secbuf::Buffer;
748    /// # use secbuf::BufferError;

754    /// assert_eq!(buf.memory_overhead(), 512); // Half overhead
755    /// # Ok::<(), BufferError>(())
756    /// ```
757    pub fn memory_overhead(&self) -> usize {
758        self.data.capacity().saturating_sub(self.len)
759    }

761    /// Creates an explicit, unsecure clone of this buffer.
762    ///
763    /// This creates a deep copy of the buffer data. Use with caution as it
764    /// creates additional copies of potentially sensitive data that must be
765    /// separately managed.
766    ///
767    /// # Security Warning
768    ///
769    /// The cloned buffer is independent and will be zeroed separately on drop.
770    /// Ensure you track all clones properly to avoid leaving sensitive data in memory.
771    ///
772    /// # Examples
773    ///
774    /// ```
775    /// use secbuf::Buffer;
776    ///
777    /// let buf1 = Buffer::new(1024);
778    /// let buf2 = buf1.clone_unsecure();
779    /// assert_eq!(buf1.capacity(), buf2.capacity());
780    /// ```
781    pub fn clone_unsecure(&self) -> Self {
782        Self {
783            data: self.data.clone(),
784            pos: self.pos,
785            len: self.len,
786        }
787    }

826    /// Resizes the buffer, preserving existing data.
827    ///
828    /// If shrinking, data beyond `new_size` is securely zeroed before release.
829    ///
830    /// # Errors
831    ///
832    /// Returns [`BufferError::SizeTooBig`] if `new_size` exceeds [`BUF_MAX_SIZE`].
833    ///
834    /// # Examples
835    ///
836    /// ```
837    /// use secbuf::Buffer;
838    /// # use secbuf::BufferError;
839    ///
840    /// let mut buf = Buffer::new(1024);
841    /// buf.resize(2048)?;
842    /// assert_eq!(buf.capacity(), 2048);
843    /// # Ok::<(), BufferError>(())
844    /// ```
845    pub fn resize(&mut self, new_size: usize) -> Result<()> {
846        if new_size > BUF_MAX_SIZE {
847            return Err(BufferError::SizeTooBig);
848        }
849
850        // If shrinking, zero the memory being released
851        if new_size < self.data.len() {
852            self.data[new_size..].zeroize();
853        }
854
855        self.data.resize(new_size, 0);
856        self.len = self.len.min(new_size);
857        self.pos = self.pos.min(new_size);
858        Ok(())
859    }

862    ///

869    ///

871    /// buf.reserve(1000);
872    /// assert!(buf.capacity() >= 1100);
873    /// ```
874    #[inline]

876        self.data.reserve(additional);
877    }

879    /// Shrinks the buffer capacity to fit the current length.
880    ///
881    /// Frees unused memory. Data being released is securely zeroed.
882    ///
883    /// # Examples
884    ///
885    /// ```
886    /// use secbuf::Buffer;
887    /// # use secbuf::BufferError;
888    ///
889    /// let mut buf = Buffer::new(1024);
890    /// buf.put_bytes(b"hello")?;
891    /// buf.shrink_to_fit();
892    /// assert_eq!(buf.capacity(), 5);
893    /// # Ok::<(), BufferError>(())
894    /// ```
895    #[inline]
896    pub fn shrink_to_fit(&mut self) {
897        // Zero unused portion before shrinking
898        if self.len < self.data.len() {
899            self.data[self.len..].zeroize();
900        }
901
902        self.data.truncate(self.len);
903        self.data.shrink_to_fit();
904    }
905}
