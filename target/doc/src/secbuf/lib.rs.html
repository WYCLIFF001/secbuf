2//! # High-Performance Buffer Library
3//!
4//! A zero-copy, lock-free buffer management library optimized for network protocols,
5//! data serialization, and high-throughput applications.
6//!
7//! Features:
8//! - Secure memory zeroing using `zeroize` crate (compiler-resistant)
9//! - Connection-scoped lifecycle management with automatic cleanup on drop
10//! - Zero-copy circular buffers for streaming I/O
11//! - Thread-safe buffer pooling with lock-free design
12//! - Optional SIMD acceleration for bulk operations
13//! - Automatic secure erasure on connection termination (mimics Dropbear behavior)
14

49#[cfg(test)]
50mod tests {
51    use super::prelude::*;

53    #[test]
54    fn test_basic_buffer() {
55        let mut buf = Buffer::new(1024);
56        buf.put_u32(42).unwrap();
57        buf.put_byte(0xFF).unwrap();
58
59        buf.set_pos(0).unwrap();
60        assert_eq!(buf.get_u32().unwrap(), 42);
61        assert_eq!(buf.get_byte().unwrap(), 0xFF);
62    }
63
64    #[test]
65    fn test_standard_pool() {
66        let pool = BufferPool::new(PoolConfig {
67            buffer_size: 1024,
68            max_pool_size: 100,
69            min_pool_size: 10,
70        });
71
72        let mut buffers = Vec::new();
73        for i in 0..50 {
74            let mut buf = pool.acquire();
75            buf.put_u32(i).unwrap();
76            buffers.push(buf);
77        }
78
79        drop(buffers);
80
81        let stats = pool.stats();
82        assert!(stats.total_acquired >= 50);
83    }

106    #[test]
107    fn test_circular_buffer() {
108        let mut ring = CircularBuffer::new(256);
109
110        ring.write(b"Chunk 1 | ").unwrap();
111        ring.write(b"Chunk 2 | ").unwrap();
112        ring.write(b"Chunk 3").unwrap();
113
114        let mut output = vec![0u8; ring.used()];
115        ring.read(&mut output).unwrap();
116
117        assert_eq!(&output, b"Chunk 1 | Chunk 2 | Chunk 3");
118    }

120    #[test]
121    fn test_unchecked_ops() {
122        let mut buf = Buffer::new(1024);
123
124        buf.put_u32(0x12345678).unwrap();
125        buf.put_u64(0xDEADBEEFCAFEBABE).unwrap();
126
127        unsafe {
128            buf.put_u32_unchecked(0xABCDEF00);
129        }
130
131        buf.set_pos(0).unwrap();
132        assert_eq!(buf.get_u32().unwrap(), 0x12345678);
133        assert_eq!(buf.get_u64().unwrap(), 0xDEADBEEFCAFEBABE);
134        assert_eq!(buf.get_u32().unwrap(), 0xABCDEF00);
135    }
